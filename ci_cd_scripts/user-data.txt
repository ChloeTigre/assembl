#!/bin/sh
sudo apt-get update -y
sudo apt-get upgrade -y

sudo addgroup assembl_group
sudo adduser assembl_user
expect "Enter new UNIX password: "
send "pass"
sudo usermod -G www-data -G assembl_group assembl_user

#sudo apt-get install -y python3-distutils
curl https://bootstrap.pypa.io/get-pip.py -o get-pip.py
sudo -H python2.7 get-pip.py
sudo pip install -U pip
sudo -H pip install virtualenv

sudo mkdir -p /home/assembl_user/test/venv
sudo virtualenv -p /usr/bin/python2.7 /home/assembl_user/test/venv/
cd /home/assembl_user/test
chown -R assembl_user:assembl_user venv/

sudo -u assembl_user -i
cd test
source venv/bin/activate
pip install
